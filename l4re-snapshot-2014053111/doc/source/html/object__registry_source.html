52  {
53  L4::Cap<L4::Ipc_gate> cap = L4Re::Env::env()->get_cap<L4::Ipc_gate>(service);
54  if (!cap.is_valid())
55  return cap;

57  l4_umword_t id = l4_umword_t(o);
