626 inline Path
627 Mount_tree::lookup(Path const &path, cxx::Ref_ptr<Mount_tree> *mt,
628  cxx::Ref_ptr<Mount_tree> *mp) throw()
629 {
630  cxx::Ref_ptr<Mount_tree> x(this);
631  Path p = path;
632  Path last_mp = p;
633 
634  if (mp)
635  *mp = this;
636 
637  while (1)
638  {
639  Path r = x->find(p, &x);
640 
641  if (!r)
642  {
643  if (mp)
644  return last_mp;
645 
646  if (mt)
647  *mt = x;
648 
649  return p;
650  }
651 
652  r.strip_sep();
653 
654  if (mp && x->_mount)
655  {
656  last_mp = r;
657  *mp = x;
658  }
659 
660  if (r.empty())
661  {
662  if (mt)
663  *mt = x;
664 
665  if (mp)
666  return last_mp;
667  else
668  return r;
669  }
670

672  }

693 inline int
694 Mount_tree::create_tree(cxx::Ref_ptr<Mount_tree> const &root,
695  char const *path, cxx::Ref_ptr<File> const &dir) throw()
696 {
697  cxx::Ref_ptr<Mount_tree> base;
698  Path p = root->lookup(Path(path), &base);
699 
700  while (!p.empty())
701  {
702  Path f = p.strip_first();
703 
704  if (f.empty())
705  return -EEXIST;
706 
707  char *name = strndup(f.path(), f.length());
708  if (!name)
709  return -ENOMEM;
710 
711  cxx::Ref_ptr<Mount_tree> nt(new Real_mount_tree(name));
712  if (!nt)
713  {
714  free(name);
715  return -ENOMEM;
716  }
717 
718  nt->_sib = base->_cld;
719  base->_cld = nt;
720 
721  base = nt;
722 
723  if (p.empty())
724  {
725  nt->_mount = dir;
726  return 0;
727  }
728  }
729 
730  return -EINVAL;
731 }
