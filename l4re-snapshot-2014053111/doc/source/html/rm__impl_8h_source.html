90 int
91 Rm::detach(l4_addr_t addr, unsigned long size, L4::Cap<Dataspace> *mem,
92  L4::Cap<L4::Task> task, unsigned flags) const throw()
93 {
94  L4::Ipc::Iostream io(l4_utcb());
95  io << Opcode(Rm_::Detach) << addr << size << flags;
96  long err = l4_error(io.call(cap(), L4Re::Protocol::Rm));
97  if (EXPECT_FALSE(err < 0))
98  return err;
99 
100  l4_addr_t start;
101  unsigned long rsize;
102  io >> start >> rsize;
103 
104  if (mem)
105  io >> *reinterpret_cast<l4_cap_idx_t*>(mem);
106 
107  if (!task.is_valid())
108  return err;
109 
110  rsize = l4_round_page(rsize);
111  unsigned order = L4_LOG2_PAGESIZE;
112  unsigned long sz = (1UL << order);
113  for (unsigned long p = start; rsize; p += sz, rsize -= sz)
114  {
115  while (sz > rsize)
116  {
117  --order;
118  sz >>= 1;
119  }
120 
121  for (;;)
122  {
123  unsigned long m = sz << 1;
124  if (m > rsize)
125  break;
126 
127  if (p & (m - 1))
128  break;
129 
130  ++order;
131  sz <<= 1;
132  }
133 
134  task->unmap(l4_fpage(p, order, L4_FPAGE_RWX),
135  L4_FP_ALL_SPACES);
136  }
137 
138  return err;
139 }
