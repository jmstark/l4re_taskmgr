230 L4_INLINE void
231 l4util_set_bit(int b, volatile l4_umword_t * dest)
232 {
233  l4_umword_t oldval, newval;
234 
235  dest += b / (sizeof(*dest) * 8); /* advance dest to the proper element */
236  b &= sizeof(*dest) * 8 - 1; /* modulo; cut off all upper bits */
237 
238  do
239  {
240  oldval = *dest;
241  newval = oldval | (1 << b);
242  }
243  while (!l4util_cmpxchg(dest, oldval, newval));
244 }
245 #endif

335 {
336  int i;
337 
338  if (!word)
339  return -1;
340

344 
345  return -1;

366 #ifndef __L4UTIL_BITOPS_HAVE_ARCH_FIND_FIRST_ZERO_BIT
367 L4_INLINE int
368 l4util_find_first_zero_bit(const void * dest, l4_size_t size)
369 {
370  l4_size_t i, j;
371  unsigned long *v = (unsigned long*)dest;
372 
373  if (!size)
374  return 0;
375 
376  size = (size + 31) & ~0x1f; /* Grmbl: adapt to x86 implementation... */
377 
378  for (i = j = 0; i < size; i++, j++)
379  {
380  if (j >= sizeof(*v) * 8)
381  {
382  j = 0;
383  v++;
384  }
385  if (!((1 << j) & *v))
386  return i;
387  }
388  return size + 1;
389 }
390 #endif
