121 void
122 List_alloc::check_overlap(void *b, unsigned long s)
123 {
124  unsigned long const mb_align = (1UL << arith::Ld<sizeof(Mem_block)>::value) - 1;
125  if ((unsigned long)b & mb_align)
126  {
127  L4::cerr << "List_alloc(FATAL): trying to free unaligned memory: "
128  << b << " align=" << arith::Ld<sizeof(Mem_block)>::value << "\n";
129  }
130 
131  Mem_block *c = _first;
132  for (;c ; c = c->next)
133  {
134  unsigned long x_s = (unsigned long)b;
135  unsigned long x_e = x_s + s;
136  unsigned long b_s = (unsigned long)c;
137  unsigned long b_e = b_s + c->size;
138 
139  if ( (x_s >= b_s && x_s < b_e)
140  || (x_e > b_s && x_e <= b_e)
141  || (b_s >= x_s && b_s < x_e)
142  || (b_e > x_s && b_e <= x_e))
143  {
144  L4::cerr << "List_alloc(FATAL): trying to free memory that "
145  "is already free: \n ["
146  << (void*)x_s << '-' << (void*)x_e << ") overlaps ["
147  << (void*)b_s << '-' << (void*)b_e << ")\n";
148  }
149  }
150 }

177 void
178 List_alloc::merge()
179 {
180  List_alloc_sanity_guard __attribute__((unused)) guard(this, __func__);
181  Mem_block *c = _first;
182  while (c && c->next)
183  {
184  unsigned long f_start = (unsigned long)c;
185  unsigned long f_end = f_start + c->size;
186  unsigned long n_start = (unsigned long)c->next;
187 
188  if (f_end == n_start)
189  {
190  c->size += c->next->size;
191  c->next = c->next->next;
192  continue;
193  }
194 
195  c = c->next;
196  }
197 }

199 void
200 List_alloc::free(void *block, unsigned long size, bool initial_free)
201 {
202  List_alloc_sanity_guard __attribute__((unused)) guard(this, __func__);
203 
204  unsigned long const mb_align = (1UL << arith::Ld<sizeof(Mem_block)>::value) - 1;
205 
206  if (initial_free)
207  {
208  // enforce alignment constraint on initial memory
209  block = (void*)(((unsigned long)block + mb_align) & ~mb_align);
210  size &= ~mb_align;
211  }
212  else
213  // blow up size to the minimum aligned size
214  size = (size + mb_align) & ~mb_align;
215 
216  check_overlap(block, size);
217 
218  Mem_block **c = &_first;
219  Mem_block *next = 0;
220 
221  if (*c)
222  {
223  while (*c && *c < block)
224  c = &(*c)->next;
225 
226  next = *c;
227  }
228 
229  *c = (Mem_block*)block;
230 
231  (*c)->next = next;
232  (*c)->size = size;
233 
234  merge();
235 }

237 void *
238 List_alloc::alloc(unsigned long size, unsigned align)
239 {
240  List_alloc_sanity_guard __attribute__((unused)) guard(this, __func__);
241 
242  unsigned long const mb_align = (1UL << arith::Ld<sizeof(Mem_block)>::value) - 1;
243 
244  // blow up size to the minimum aligned size
245  size = (size + mb_align) & ~mb_align;
246 
247  unsigned long almask = align ? (align -1) : 0;
248 
249  // minimum alignment is given by the size of a Mem_block
250  if (almask < mb_align)
251  almask = mb_align;
252 
253  Mem_block **c = &_first;
254 
255  for (; *c; c=&(*c)->next)
256  {
257  // address of free memory block
258  unsigned long n_start = (unsigned long)(*c);
259 
260  // block too small, next
261  // XXX: maybe we can skip this and just do the test below
262  if ((*c)->size < size)
263  continue;
264 
265  // aligned start address within the free block
266  unsigned long a_start = (n_start + almask) & ~almask;
267 
268  // hm, block too small after alignment, next
269  if (a_start - n_start >= (*c)->size)
270  continue;
271 
272  // remaining size after subtracting the padding
273  // for the alignment
274  unsigned long r_size = (*c)->size - a_start + n_start;
275 
276  // block too small
277  if (r_size < size)
278  continue;
279 
280  if (a_start < n_start)
281  {
282  // have free space before the allocated block
283  // shrink the block and set c to the next pointer of that
284  // block
285  (*c)->size -= r_size;
286  c = &(*c)->next;
287  }
288  else
289  // drop the block, c remains the next pointer of the
290  // previous block
291  *c = (*c)->next;
292 
293  // allocated the whole remaining space
294  if (r_size == size)
295  return (void*)a_start;
296 
297  // add a new free block behind the allocated block
298  Mem_block *m = (Mem_block*)(a_start + size);
299  m->next = *c;
300  m->size = r_size - size;
301  *c = m;
302  return (void*)a_start;
303  }
304 
305  return 0;
306 }
