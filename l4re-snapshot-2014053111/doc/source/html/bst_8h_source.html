253 {
254  Dir d;
255  Bst_node *r = 0;
256 
257  for (Bst_node *q = _head; q; q = Bst_node::next(q, d))
258  {
259  d = dir(key, q);
260  if (d == Dir::L)
261  r = q; // found a node greater than key

263  return static_cast<Node*>(q);

281  return Iterator(static_cast<Node*>(q), static_cast<Node *>(r));
