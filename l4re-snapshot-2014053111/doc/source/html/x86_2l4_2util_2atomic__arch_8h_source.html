30 /* atomic compare and exchange 64 bit value */
31 #define __L4UTIL_ATOMIC_HAVE_ARCH_CMPXCHG64
32 L4_INLINE int
33 l4util_cmpxchg64(volatile l4_uint64_t * dest,
34  l4_uint64_t cmp_val, l4_uint64_t new_val)
35 {
36  unsigned char ret;
37  l4_umword_t dummy;
38 
39  __asm__ __volatile__
40  (
41 #ifdef __PIC__
42  LOCK "xchg %%esi,%%ebx\n\t"
43 #endif
44  LOCK "cmpxchg8b %5\n\t"
45  "sete %0\n\t"
46 #ifdef __PIC__
47  "xchg %%esi,%%ebx\n\t"
48 #endif
49  :
50  "=a" (ret), /* return val, 0 or 1 */
51  "=d" (dummy)
52  :
53  "A" (cmp_val),
54  "c" ((unsigned int)(new_val>>32ULL)),
55 #ifdef __PIC__
56  "S"
57 #else
58  "b"
59 #endif
60  ((unsigned int)new_val),
61  "m" (*dest) /* 3 mem, destination operand */
62  :
63  "memory", "cc"
64  );
65 
66  return ret;
67 }

70 #define __L4UTIL_ATOMIC_HAVE_ARCH_CMPXCHG32
71 L4_INLINE int
72 l4util_cmpxchg32(volatile l4_uint32_t * dest,
73  l4_uint32_t cmp_val, l4_uint32_t new_val)
74 {
75  l4_uint32_t tmp;
76 
77  __asm__ __volatile__
78  (
79  LOCK "cmpxchgl %1, %3 \n\t"
80  :
81  "=a" (tmp) /* 0 EAX, return val */
82  :
83  "r" (new_val), /* 1 reg, new value */
84  "0" (cmp_val), /* 2 EAX, compare value */
85  "m" (*dest) /* 3 mem, destination operand */
86  :
87  "memory", "cc"
88  );
