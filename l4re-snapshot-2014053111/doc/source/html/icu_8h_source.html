326 L4_INLINE l4_msgtag_t

350 L4_INLINE l4_msgtag_t
351 l4_icu_info_u(l4_cap_idx_t icu, l4_icu_info_t *info,
352  l4_utcb_t *utcb) L4_NOTHROW
353 {
354  l4_msgtag_t res;
355  l4_msg_regs_t *m = l4_utcb_mr_u(utcb);
356  m->mr[0] = L4_ICU_OP_INFO;
357  res = l4_ipc_call(icu, utcb, l4_msgtag(L4_PROTO_IRQ, 1, 0, 0), L4_IPC_NEVER);
358  if (!l4_msgtag_has_error(res) && l4_msgtag_label(res) >= 0
359  && l4_msgtag_words(res) >= 3)
360  {
361  info->features = m->mr[0];
362  info->nr_irqs = m->mr[1];
363  info->nr_msis = m->mr[2];
364  }
365  return res;
366 }
