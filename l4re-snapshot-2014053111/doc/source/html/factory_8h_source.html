262 L4_INLINE l4_msgtag_t
263 l4_factory_create_gate_u(l4_cap_idx_t factory,
264  l4_cap_idx_t target_cap,
265  l4_cap_idx_t thread_cap, l4_umword_t label,
266  l4_utcb_t *u) L4_NOTHROW
267 {
268  l4_msgtag_t t;
269  l4_msg_regs_t *v;
270  int items = 0;
271  t = l4_factory_create_start_u(0, target_cap, u);
272  l4_factory_create_add_uint_u(label, &t, u);
273  v = l4_utcb_mr_u(u);
274  if (!(thread_cap & L4_INVALID_CAP_BIT))
275  {
276  items = 1;
277  v->mr[3] = l4_map_obj_control(0,0);
278  v->mr[4] = l4_obj_fpage(thread_cap, 0, L4_FPAGE_RWX).raw;
279  }
280  t = l4_msgtag(l4_msgtag_label(t), l4_msgtag_words(t), items, l4_msgtag_flags(t));
281  return l4_factory_create_commit_u(factory, t, u);
282 }

361 L4_INLINE int
362 l4_factory_create_add_fpage_u(l4_fpage_t d, l4_msgtag_t *tag,
363  l4_utcb_t *u) L4_NOTHROW
364 {
365  l4_msg_regs_t *v = l4_utcb_mr_u(u);
366  int w = l4_msgtag_words(*tag);
367  if (w + 2 > L4_UTCB_GENERIC_DATA_SIZE)
368  return 0;
369  v->mr[w] = L4_VARG_TYPE_FPAGE | (sizeof(l4_fpage_t) << 16);
370  v->mr[w + 1] = d.raw;
371  w += 2;
372  tag->raw = (tag->raw & ~0x3fUL) | (w & 0x3f);
373  return 1;
374 }

413 L4_INLINE int
414 l4_factory_create_add_lstr_u(char const *s, int len, l4_msgtag_t *tag,
415  l4_utcb_t *u) L4_NOTHROW
416 {
417 
418  l4_msg_regs_t *v = l4_utcb_mr_u(u);
419  int w = l4_msgtag_words(*tag);
420  char *c;
421  int i;
422 
423  if (w + 1 + (len + sizeof(l4_umword_t) - 1) / sizeof(l4_umword_t)
424  > L4_UTCB_GENERIC_DATA_SIZE)
425  return 0;
426 
427  v->mr[w] = L4_VARG_TYPE_STRING | (len << 16);
428  c = (char*)&v->mr[w + 1];
429  for (i = 0; i < len; ++i)
430  *c++ = *s++;
431 
432  w = w + 1 + (len + sizeof(l4_umword_t) - 1) / sizeof(l4_umword_t);
433 
434  tag->raw = (tag->raw & ~0x3fUL) | (w & 0x3f);
435  return 1;
436 }
